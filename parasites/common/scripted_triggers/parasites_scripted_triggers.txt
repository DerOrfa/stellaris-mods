pop_can_be_host = {
	is_organic_species = yes
	OR = { # not parasite itself, or enslaved parasite, or in labor camp
		NOT = { has_trait = trait_parasite }
		AND = {
			is_enslaved = yes
			NOT = { has_slavery_type = { type = slavery_military } }
		}
		AND = {
			has_citizenship_type = { type = citizenship_purge }
			has_purge_type = { type = purge_labor_camps }
		}
	}
}

is_parasite_leader = {
	OR = {
		has_leader_flag = parasite_leader 
		species = { has_species_flag = parasite_speciation_1 }
	}
}
